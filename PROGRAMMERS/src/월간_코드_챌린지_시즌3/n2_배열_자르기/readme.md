import java.util.*;
class Solution {
    public int[] solution(int n, long left, long right) {
        int[] answer = new int[(int)(right-left+1)];
        int idx = 0;
        for(int i=0; i<answer.length; i++){
            int num = Math.max((int)((left+i)%n),(int)((left+i)/n));
            num += 1;
            answer[idx++] = num;
        }
        return answer;
    }
}
